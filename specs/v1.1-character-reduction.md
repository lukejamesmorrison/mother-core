# MotherCore v1.1 - Character Reduction Specification

## Overview

This specification documents the analysis and implementation plan for reducing the minified character count of MotherOS (MotherCore + extension modules). The goal is to improve code reuse, reduce repetitive patterns, and maximize the available character budget within Space Engineers' 100,000 character limit.

**Current Status:** 99,799 characters (99.8% capacity)

## Problem Statement

The current codebase contains significant repetition in command implementations. Most block-targeting commands follow an identical pattern:

```csharp
public override string Execute(TerminalCommand command)
{
    if (command.Arguments.Count == 0)
        return CommandBus.Messages.NoArgumentsProvided;
    
    string blockName = command.Arguments[0];
    List<TBlock> blocks = Module.GetBlocksByName<TBlock>(blockName);
    
    if (blocks.Count == 0)
        return MessageFormatter.Format(BlockMessages.BlockNotFound, blockName);
    
    blocks.ForEach(block => Module.DoSomething(block));
    
    return MessageFormatter.Format(BlockMessages.SomeMessage, blockName);
}
```

This pattern is repeated in **35+ command classes**, each requiring ~55 lines of code.

## Solution: Generic Block Command Base Classes

### Base Class 1: `BlockCommand<TBlock, TModule>`

For commands that operate on a single block type.

```csharp
// MotherCore\src\Commands\BlockCommand.cs
using Sandbox.ModAPI.Ingame;
using System.Collections.Generic;

namespace IngameScript
{
    /// <summary>
    /// Generic base class for commands that operate on blocks by name.
    /// Eliminates boilerplate for the common pattern of:
    /// 1. Get block name from arguments
    /// 2. Find blocks by name
    /// 3. Execute action on each block
    /// 4. Return result message
    /// </summary>
    /// <typeparam name="TBlock">The block interface type</typeparam>
    /// <typeparam name="TModule">The module type that owns this command</typeparam>
    public abstract class BlockCommand<TBlock, TModule> : BaseModuleCommand
        where TBlock : class, IMyTerminalBlock
        where TModule : BaseModule
    {
        protected readonly TModule Module;
        protected abstract string SuccessMessage { get; }

        protected BlockCommand(TModule module) => Module = module;

        protected abstract void ExecuteOnBlock(TBlock block, TerminalCommand command);

        public override string Execute(TerminalCommand command)
        {
            if (command.Arguments.Count == 0)
                return CommandBus.Messages.NoArgumentsProvided;

            var blockName = command.Arguments[0];
            var blocks = Module.GetBlocksByName<TBlock>(blockName);

            if (blocks.Count == 0)
                return MessageFormatter.Format(BlockMessages.BlockNotFound, blockName);

            foreach (var block in blocks) ExecuteOnBlock(block, command);

            return MessageFormatter.Format(SuccessMessage, blockName);
        }
    }
}
```

### Base Class 2: `DualBlockCommand<T1, T2, TModule>`

For commands that operate on two block types (e.g., `IMyLightingBlock` and `IMySearchlight`).

```csharp
// MotherCore\src\Commands\DualBlockCommand.cs
using Sandbox.ModAPI.Ingame;

namespace IngameScript
{
    /// <summary>
    /// Generic base class for commands that operate on two block types.
    /// Used for blocks like lights where IMyLightingBlock and IMySearchlight
    /// are separate interfaces.
    /// </summary>
    public abstract class DualBlockCommand<TBlock1, TBlock2, TModule> : BaseModuleCommand
        where TBlock1 : class, IMyTerminalBlock
        where TBlock2 : class, IMyTerminalBlock
        where TModule : BaseModule
    {
        protected readonly TModule Module;
        protected abstract string SuccessMessage { get; }

        protected DualBlockCommand(TModule module) => Module = module;

        protected abstract void ExecuteOnBlock(TBlock1 block, TerminalCommand command);
        protected abstract void ExecuteOnBlock(TBlock2 block, TerminalCommand command);

        public override string Execute(TerminalCommand command)
        {
            if (command.Arguments.Count == 0)
                return CommandBus.Messages.NoArgumentsProvided;

            var blockName = command.Arguments[0];
            var primaryBlocks = Module.GetBlocksByName<TBlock1>(blockName);
            var secondaryBlocks = Module.GetBlocksByName<TBlock2>(blockName);

            if (primaryBlocks.Count + secondaryBlocks.Count == 0)
                return MessageFormatter.Format(BlockMessages.BlockNotFound, blockName);

            foreach (var block in primaryBlocks) ExecuteOnBlock(block, command);
            foreach (var block in secondaryBlocks) ExecuteOnBlock(block, command);

            return MessageFormatter.Format(SuccessMessage, blockName);
        }
    }
}
```

## Refactoring Examples

### Example 1: Simple Command (Tier 1)

**Before (~55 lines):**
```csharp
public class OpenDoorCommand : BaseModuleCommand
{
    readonly DoorModule Module;
    public override string Name => "door/open";
    
    public OpenDoorCommand(DoorModule module) { Module = module; }
    
    public override string Execute(TerminalCommand command)
    {
        if (command.Arguments.Count == 0)
            return CommandBus.Messages.NoArgumentsProvided;
        
        string doorName = command.Arguments[0];
        List<IMyDoor> doors = Module.GetBlocksByName<IMyDoor>(doorName);
        
        if (doors.Count == 0)
            return MessageFormatter.Format(BlockMessages.BlockNotFound, doorName);
        
        doors.ForEach(door => Module.OpenDoor(door));
        return MessageFormatter.Format(BlockMessages.BlockOpen, doorName);
    }
}
```

**After (~12 lines):**
```csharp
public class OpenDoorCommand : BlockCommand<IMyDoor, DoorModule>
{
    public override string Name => "door/open";
    protected override string SuccessMessage => BlockMessages.BlockOpen;
    
    public OpenDoorCommand(DoorModule module) : base(module) { }
    
    protected override void ExecuteOnBlock(IMyDoor door, TerminalCommand command) => Module.OpenDoor(door);
}
```

### Example 2: Command with Options (Tier 2)

**Before:**
```csharp
public class LockRotorCommand : BaseModuleCommand
{
    readonly RotorModule Module;
    public override string Name => "rotor/lock";
    
    public LockRotorCommand(RotorModule module) { Module = module; }
    
    public override string Execute(TerminalCommand command)
    {
        if (command.Arguments.Count == 0)
            return CommandBus.Messages.NoArgumentsProvided;
        
        string rotorName = command.Arguments[0];
        List<IMyMotorStator> rotors = Module.GetBlocksByName<IMyMotorStator>(rotorName);
        
        if (rotors.Count == 0)
            return MessageFormatter.Format(BlockMessages.BlockNotFound, rotorName);
        
        bool stop = command.GetOption("stop") != null && command.GetOption("stop") != "false";
        rotors.ForEach(rotor => Module.Lock(rotor, stop));
        
        return MessageFormatter.Format(BlockMessages.BlockLocked, rotorName);
    }
}
```

**After:**

public class LockRotorCommand : BlockCommand<IMyMotorStator, RotorModule>

    public override string Name => "rotor/lock";

    protected override void ExecuteOnBlock(IMyMotorStator rotor, TerminalCommand command)
    {
        bool stop = command.GetOption("stop") != null && command.GetOption("stop") != "false";
        Module.Lock(rotor, stop);
    }

### Example 3: Dual Block Command (Tier 3)

**Before (~63 lines):**
```csharp
public class SetLightColorCommand : BaseModuleCommand
{
    readonly LightModule Module;
    public override string Name => "light/color";
    
    public SetLightColorCommand(LightModule module) { Module = module; }

    public override string Execute(TerminalCommand command)
    {
        if (command.Arguments.Count == 0)
            return CommandBus.Messages.NoArgumentsProvided;
               
        string lightName = command.Arguments[0];
        string colorString = command.Arguments[1];

        List<IMyLightingBlock> lights = Module.GetBlocksByName<IMyLightingBlock>(lightName);
        List<IMySearchlight> searchlights = Module.GetBlocksByName<IMySearchlight>(lightName);

        if ((lights.Count + searchlights.Count) == 0)
            return MessageFormatter.Format(BlockMessages.BlockNotFound, lightName);

        Color color = ColorHelper.GetColor(colorString);

        if (color != null)
        {
            lights.ForEach(light => Module.SetColor(light, color));
            searchlights.ForEach(light => Module.SetColor(light, color));
            return MessageFormatter.Format(BlockMessages.BlockUpdated, lightName, $"color={color}");
        }
        return "Invalid color format.";
    }
}
```

**After (~25 lines):**
```csharp
public class SetLightColorCommand : DualBlockCommand<IMyLightingBlock, IMySearchlight, LightModule>
{
    public override string Name => "light/color";
    protected override string SuccessMessage => BlockMessages.BlockUpdated;

    public SetLightColorCommand(LightModule module) : base(module) { }

    Color parsedColor;

    public override string Execute(TerminalCommand command)
    {
        if (command.Arguments.Count < 2)
            return CommandBus.Messages.NoArgumentsProvided;

        parsedColor = ColorHelper.GetColor(command.Arguments[1]);
        if (parsedColor == null) return "Invalid color format.";

        return base.Execute(command);
    }

    protected override void ExecuteOnBlock(IMyLightingBlock light, TerminalCommand command) => Module.SetColor(light, parsedColor);
    protected override void ExecuteOnBlock(IMySearchlight searchlight, TerminalCommand command) => Module.SetColor(searchlight, parsedColor);
}
```
